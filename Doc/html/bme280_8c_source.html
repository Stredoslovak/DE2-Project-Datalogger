414{

417
418 /* Read the chip-id of bme280 sensor */
419 rslt = bme280_get_regs(BME280_REG_CHIP_ID, &chip_id, 1, dev);
420
421 /* Check for chip id validity */
422 if (rslt == BME280_OK)
423 {
424 if (chip_id == BME280_CHIP_ID)
425 {

427
428 /* Reset the sensor */

433 /* Read the calibration data */

450{
451 int8_t rslt;
452
453 /* Check for null pointer in the device structure */
454 rslt = null_ptr_check(dev);
455
456 if ((rslt == BME280_OK) && (reg_data != NULL))
457 {
458 /* If interface selected is SPI */

469 {

485int8_t bme280_set_regs(uint8_t *reg_addr, const uint8_t *reg_data, uint32_t len, struct bme280_dev *dev)
486{
487 int8_t rslt;
488 uint8_t temp_buff[20]; /* Typically not to write more than 10 registers */
489 uint32_t temp_len;
490 uint32_t reg_addr_cnt;
491
492 if (len > BME280_MAX_LEN)
493 {
494 len = BME280_MAX_LEN;
495 }
496
497 /* Check for null pointer in the device structure */
498 rslt = null_ptr_check(dev);
499
500 /* Check for arguments validity */
501 if ((rslt == BME280_OK) && (reg_addr != NULL) && (reg_data != NULL))
502 {
503 if (len != 0)
504 {
505 temp_buff[0] = reg_data[0];
506
507 /* If interface selected is SPI */
508 if (dev->intf != BME280_I2C_INTF)
509 {
510 for (reg_addr_cnt = 0; reg_addr_cnt < len; reg_addr_cnt++)
511 {
512 reg_addr[reg_addr_cnt] = reg_addr[reg_addr_cnt] & 0x7F;
513 }
514 }
515
516 /* Burst write mode */
517 if (len > 1)
518 {
519 /* Interleave register address w.r.t data for
520 * burst write
521 */
522 interleave_reg_addr(reg_addr, temp_buff, reg_data, len);
523 temp_len = ((len * 2) - 1);
524 }
525 else
526 {
527 temp_len = len;
528 }
529
530 dev->intf_rslt = dev->write(reg_addr[0], temp_buff, temp_len, dev->intf_ptr);
531
532 /* Check for communication error */
533 if (dev->intf_rslt != BME280_INTF_RET_SUCCESS)
534 {
535 rslt = BME280_E_COMM_FAIL;
536 }
537 }
538 else
539 {
540 rslt = BME280_E_INVALID_LEN;
541 }
542 }
543 else
544 {
545 rslt = BME280_E_NULL_PTR;
546 }
547
548 return rslt;
549}

677int8_t bme280_soft_reset(struct bme280_dev *dev)
678{
679 int8_t rslt;
680 uint8_t reg_addr = BME280_REG_RESET;
681 uint8_t status_reg = 0;
682 uint8_t try_run = 5;
683
684 /* 0xB6 is the soft reset command */
685 uint8_t soft_rst_cmd = BME280_SOFT_RESET_COMMAND;
686
687 /* Write the soft reset command in the sensor */
688 rslt = bme280_set_regs(&reg_addr, &soft_rst_cmd, 1, dev);
689
690 if (rslt == BME280_OK)
691 {
692 /* If NVM not copied yet, Wait for NVM to copy */
693 do
694 {
695 /* As per data sheet - Table 1, startup time is 2 ms. */
696 dev->delay_us(BME280_STARTUP_DELAY, dev->intf_ptr);
697 rslt = bme280_get_regs(BME280_REG_STATUS, &status_reg, 1, dev);
698
699 } while ((rslt == BME280_OK) && (try_run--) && (status_reg & BME280_STATUS_IM_UPDATE));
700
701 if (status_reg & BME280_STATUS_IM_UPDATE)
702 {
703 rslt = BME280_E_NVM_COPY_FAILED;
704 }
705 }
706
707 return rslt;
708}

737 */
738 rslt = bme280_compensate_data(sensor_comp, &uncomp_data, comp_data, &dev->calib_data);
739 }
740 }

754int8_t bme280_compensate_data(uint8_t sensor_comp,
755 const struct bme280_uncomp_data *uncomp_data,
756 struct bme280_data *comp_data,
757 struct bme280_calib_data *calib_data)
758{
759 int8_t rslt = BME280_OK;
760
761 if ((uncomp_data != NULL) && (comp_data != NULL) && (calib_data != NULL))
762 {
763 /* Initialize to zero */
764 comp_data->temperature = 0;
765 comp_data->pressure = 0;
766 comp_data->humidity = 0;
767
768 /* If pressure or temperature component is selected */
769 if (sensor_comp & (BME280_PRESS | BME280_TEMP | BME280_HUM))
770 {
771 /* Compensate the temperature data */
772 comp_data->temperature = compensate_temperature(uncomp_data, calib_data);
773 }
774
775 if (sensor_comp & BME280_PRESS)
776 {
777 /* Compensate the pressure data */
778 comp_data->pressure = compensate_pressure(uncomp_data, calib_data);
779 }
780
781 if (sensor_comp & BME280_HUM)
782 {
783 /* Compensate the humidity data */
784 comp_data->humidity = compensate_humidity(uncomp_data, calib_data);
785 }
786 }
787 else
788 {
789 rslt = BME280_E_NULL_PTR;
790 }
791
792 return rslt;

799int8_t bme280_cal_meas_delay(uint32_t *max_delay, const struct bme280_settings *settings)
800{
801 int8_t rslt = BME280_OK;
802 uint8_t temp_osr;
803 uint8_t pres_osr;
804 uint8_t hum_osr;
805
806 /* Array to map OSR config register value to actual OSR */
807 uint8_t osr_sett_to_act_osr[] = { 0, 1, 2, 4, 8, 16 };
808
809 if ((settings != NULL) && (max_delay != NULL))
810 {
811 /* Mapping osr settings to the actual osr values e.g. 0b101 -> osr X16 */
812 if (settings->osr_t <= BME280_OVERSAMPLING_16X)
813 {
814 temp_osr = osr_sett_to_act_osr[settings->osr_t];
815 }
816 else
817 {
818 temp_osr = BME280_OVERSAMPLING_MAX;
819 }
820
821 if (settings->osr_p <= BME280_OVERSAMPLING_16X)
822 {
823 pres_osr = osr_sett_to_act_osr[settings->osr_p];
824 }
825 else
826 {
827 pres_osr = BME280_OVERSAMPLING_MAX;
828 }
829
830 if (settings->osr_h <= BME280_OVERSAMPLING_16X)
831 {
832 hum_osr = osr_sett_to_act_osr[settings->osr_h];
833 }
834 else
835 {
836 hum_osr = BME280_OVERSAMPLING_MAX;
837 }
838
839 (*max_delay) =
840 (uint32_t)((BME280_MEAS_OFFSET + (BME280_MEAS_DUR * temp_osr) +
841 ((BME280_MEAS_DUR * pres_osr) + BME280_PRES_HUM_MEAS_OFFSET) +
842 ((BME280_MEAS_DUR * hum_osr) + BME280_PRES_HUM_MEAS_OFFSET)));
843 }
844 else
845 {
846 rslt = BME280_E_NULL_PTR;
847 }
848
849 return rslt;
850}

1028static void parse_sensor_data(const uint8_t *reg_data, struct bme280_uncomp_data *uncomp_data)
1029{
1030 /* Variables to store the sensor data */
1031 uint32_t data_xlsb;
1032 uint32_t data_lsb;
1033 uint32_t data_msb;
1034
1035 /* Store the parsed register values for pressure data */
1036 data_msb = (uint32_t)reg_data[0] << BME280_12_BIT_SHIFT;
1037 data_lsb = (uint32_t)reg_data[1] << BME280_4_BIT_SHIFT;
1038 data_xlsb = (uint32_t)reg_data[2] >> BME280_4_BIT_SHIFT;
1039 uncomp_data->pressure = data_msb | data_lsb | data_xlsb;
1040
1041 /* Store the parsed register values for temperature data */
1042 data_msb = (uint32_t)reg_data[3] << BME280_12_BIT_SHIFT;
1043 data_lsb = (uint32_t)reg_data[4] << BME280_4_BIT_SHIFT;
1044 data_xlsb = (uint32_t)reg_data[5] >> BME280_4_BIT_SHIFT;
1045 uncomp_data->temperature = data_msb | data_lsb | data_xlsb;
1046
1047 /* Store the parsed register values for humidity data */
1048 data_msb = (uint32_t)reg_data[6] << BME280_8_BIT_SHIFT;
1049 data_lsb = (uint32_t)reg_data[7];
1050 uncomp_data->humidity = data_msb | data_lsb;
1051}
