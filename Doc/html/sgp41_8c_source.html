48/* Refactored SGP41 wrapper: provide init + single-measure interface so main.c

60/* Initialize SGP41 algorithm state and perform optional self-test.
61 * Caller must call `twi_init()` before this function.
62 * Returns 0 on success, non-zero on error.
63 */
64int sgp41_init_simple(void)
65{
66 if (initialized) return 0;
67
68 /* Get serial number (optional, non-fatal) */
69 uint16_t serialNumber[3];
70 (void)sgp41_getSerialNumber(serialNumber);
71
72 /* Run self-test once (non-fatal) */
73 uint16_t testResult = 0;
74 (void)sgp41_executeSelfTest(&testResult);
75
76 /* Initialize gas index algorithms for VOC and NOx */
77 GasIndexAlgorithm_init(&voc_params, GasIndexAlgorithm_ALGORITHM_TYPE_VOC);
78 GasIndexAlgorithm_init(&nox_params, GasIndexAlgorithm_ALGORITHM_TYPE_NOX);
79
80 initialized = 1;
81 return 0;
82}

84/* Perform one measurement and compute VOC/NOx indices. Caller provides pointers
85 * to store results. Returns 0 on success or non-zero on error.
86 */
87int sgp41_measure_once(int32_t *voc_index, int32_t *nox_index)
88{
89 if (!initialized) return -1;
90
91 uint16_t defaultRh = 0x8000;
92 uint16_t defaultT = 0x6666;
93 uint16_t srawVoc = 0;
94 uint16_t srawNox = 0;
95
96 uint16_t err;
97 if (conditioning_s > 0) {
98 /* run conditioning command to keep sensor heater conditioned */
99 (void)sgp41_executeConditioning(defaultRh, defaultT, &srawVoc);
100 /* always perform a measurement to obtain both VOC and NOx */
101 err = sgp41_measureRawSignals(defaultRh, defaultT, &srawVoc, &srawNox);
102 conditioning_s--;
103 } else {
104 err = sgp41_measureRawSignals(defaultRh, defaultT, &srawVoc, &srawNox);
105 }
106
107 if (err) return (int)err;
108
109 /* Process raw sraw values through gas index algorithms */
110 GasIndexAlgorithm_process(&voc_params, (int32_t)srawVoc, voc_index);
111 GasIndexAlgorithm_process(&nox_params, (int32_t)srawNox, nox_index);
112
113 return 0;
114}
