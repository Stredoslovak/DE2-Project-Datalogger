78void set_interrupt_source(void)
79{
80 if (RTC_OK == 0)
81 {
82 // RTC is OK, use external interrupt on PD2
83 tim1_ovf_disable();
84 rtc_write_reg(0x0e, 0x00); // Enable square-wave output
85 setup_ext_int();

87 else
88 {
89 // RTC not OK, use Timer1 interrupt
90 EIMSK &= ~(1 << INT0); // Disable INT0

95{
96 uart_puts("Waiting for time data (YYYY,MM,DD,HH,MM,SS,DOW)...\r\n");
97
98 char time_buffer[30];
99 uint8_t idx = 0;
100
101 while(1) {
102 uint16_t data = uart_getc();
103 if(data & UART_NO_DATA) continue;
104
105 char ch = (char)(data & 0xFF);
106
107 if(ch == '\r' || ch == '\n') break;
108
109 time_buffer[idx++] = ch;
110 if(idx >= 29) break;
111 }
112 time_buffer[idx] = '\0';
113
114 // Parse: YYYY,MM,DD,HH,MM,SS,DOW
115 int year, month, day, hour, minute, second, dow;
116 if(sscanf(time_buffer, "%d,%d,%d,%d,%d,%d,%d",
117 &year, &month, &day, &hour, &minute, &second, &dow) == 7) {
118 rtc_set_time(hour, minute, second);
119 rtc_set_date(dow, day, month, (year % 100));
120 // rtc_set_day(dow); // 1=Sunday ... 7=Saturday
121
122 const char* days[] = {"", "Sun", "Mon", "Tue", "Wed", "Thu", "Fri", "Sat"};
123 char msg[60];
124 sprintf(msg, "Time set: %04d-%02d-%02d (%s) %02d:%02d:%02d\r\n",
125 year, month, day, days[dow], hour, minute, second);
126 uart_puts(msg);
127 } else {
128 uart_puts("Error parsing time!\r\n");
129 }
