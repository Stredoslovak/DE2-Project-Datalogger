141/* Perform one forced measurement and return compensated values.
142 * t100: temperature in 0.01 degC
143 * press_pa: pressure in Pa
144 * hum_x1024: humidity in percent * 1024
145 * Returns 0 on success, non-zero on failure.
146 */
147int bme_read_once(struct bme280_dev *dev, int32_t *t100, uint32_t *press_pa, uint32_t *hum_x1024)
148{
149 /* Trigger one-shot measurement (forced) */
150 int8_t rslt = bme280_set_sensor_mode(BME280_POWERMODE_FORCED, dev);
151 if (rslt != BME280_OK) return -1;
152
153 /* Wait required measurement time */
154 struct bme280_settings settings;
155 /* read back current settings to compute delay
156 * If reading settings fails, fall back to a conservative delay
157 */
158 rslt = bme280_get_sensor_settings(&settings, dev);
159 uint32_t meas_delay_us = 0;
160 if (rslt == BME280_OK)
161 bme280_cal_meas_delay(&meas_delay_us, &settings);
162 else
163 meas_delay_us = 10000; /* 10 ms conservative */
164
165 user_delay_us(meas_delay_us, dev->intf_ptr);
166
167 /* Read raw measurement registers (8 bytes: press(3), temp(3), hum(2)) */
168 uint8_t data[8];
169 rslt = bme280_get_regs(BME280_REG_DATA, data, BME280_LEN_P_T_H_DATA, dev);
170 if (rslt != BME280_OK)
171 {
172 return -2;
173 }
174
175 uint32_t adc_P = ((uint32_t)data[0] << 12) | ((uint32_t)data[1] << 4) | ((uint32_t)data[2] >> 4);
176 uint32_t adc_T = ((uint32_t)data[3] << 12) | ((uint32_t)data[4] << 4) | ((uint32_t)data[5] >> 4);
177 uint32_t adc_H = ((uint32_t)data[6] << 8) | (uint32_t)data[7];
178
179 /* Compensate using manufacturer's integer routines */
180 *t100 = BME280_compensate_T_int32((int32_t)adc_T, &dev->calib_data); /* 0.01 degC */
181 uint32_t p_q24_8 = BME280_compensate_P_int64((int32_t)adc_P, &dev->calib_data); /* Q24.8 */
182 *hum_x1024 = bme280_compensate_H_int32((int32_t)adc_H, &dev->calib_data); /* Q22.10 */
183
184 /* Convert pressure Q24.8 -> Pa (divide by 256) */
185 *press_pa = (uint32_t)((uint64_t)p_q24_8 / 256ULL);
186
187 return 0;
188}
