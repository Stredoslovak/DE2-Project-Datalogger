14/*
15 * Function: twi_init()
16 * Purpose: Initialize TWI unit, enable internal pull-ups, and set SCL
17 * frequency.
18 * Returns: none
19 */
20void twi_init(void)
21{
22 /* Enable internal pull-up resistors */
23 DDR(TWI_PORT) &= ~((1<<TWI_SDA_PIN) | (1<<TWI_SCL_PIN));
24 TWI_PORT |= (1<<TWI_SDA_PIN) | (1<<TWI_SCL_PIN);
25
26 /* Set SCL frequency */
27 TWSR &= ~((1<<TWPS1) | (1<<TWPS0));
28 TWBR = TWI_BIT_RATE_REG;
29}

32/*
33 * Function: twi_start()
34 * Purpose: Start communication on I2C/TWI bus.
35 * Returns: none
36 */
37void twi_start(void)
38{
39 /* Send Start condition */
40 TWCR = (1<<TWINT) | (1<<TWSTA) | (1<<TWEN);
41 while ((TWCR & (1<<TWINT)) == 0);
42}

45/*
46 * Function: twi_write()
47 * Purpose: Write one byte to the I2C/TWI bus.
48 * Input: data Byte to be transmitted
49 * Returns: ACK/NACK received value
50 */
51uint8_t twi_write(uint8_t data)
52{
53 uint8_t twi_status;
54
55 /* Send SLA+R, SLA+W, or data byte on I2C/TWI bus */
56 TWDR = data;
57 TWCR = (1<<TWINT) | (1<<TWEN);
58 while ((TWCR & (1<<TWINT)) == 0);
59
60 /* Check value of TWI status register */
61 twi_status = TWSR & 0xf8;
62
63 /* Status Code:
64 - 0x18: SLA+W has been transmitted and ACK received
65 - 0x28: Data byte has been transmitted and ACK has been received
66 - 0x40: SLA+R has been transmitted and ACK received
67 */
68 if (twi_status == 0x18 || twi_status == 0x28 || twi_status == 0x40)
69 return 0; /* ACK received */
70 else
71 return 1; /* NACK received */
72}

75/*
76 * Function: twi_read()
77 * Purpose: Read one byte from the I2C/TWI bus and acknowledge
78 * it by ACK or NACK.
79 * Input: ack ACK/NACK value to be transmitted
80 * Returns: Received data byte
81 */
82uint8_t twi_read(uint8_t ack)
83{
84 if (ack == TWI_ACK)
85 TWCR = (1<<TWINT) | (1<<TWEN) | (1<<TWEA);
86 else
87 TWCR = (1<<TWINT) | (1<<TWEN);
88 while ((TWCR & (1<<TWINT)) == 0);
89
90 return (TWDR);
91}

123/*
124 * Function: twi_readfrom_mem_into()
125 * Purpose: Read into buf from the peripheral starting from the memory address.
126 * Input: addr Slave address
127 * memaddr Starting address
128 * buf Buffer to be read into
129 * nbytes Number of bytes
130 * Returns: None
131 */
132void twi_readfrom_mem_into(uint8_t addr, uint8_t memaddr, volatile uint8_t *buf, uint8_t nbytes)
133{
134 twi_start();
135 if (twi_write((addr<<1) | TWI_WRITE) == 0)
136 {
137 // Set starting address
138 twi_write(memaddr);
139 twi_stop();
140
141 // Read data into the buffer
142 twi_start();
143 twi_write((addr<<1) | TWI_READ);
144 if (nbytes >= 2)
145 {
146 for (uint8_t i=0; i<(nbytes-1); i++)
147 {
148 *buf++ = twi_read(TWI_ACK);
149 }
150 }
151 *buf = twi_read(TWI_NACK);
152 twi_stop();
153 }
154 else
155 {
156 twi_stop();
157 }

160uint8_t twi_writeto_mem(uint8_t addr, uint8_t memaddr, uint8_t data)
161{
162 twi_start();
163 /* Send SLA+W and check ACK */
164 if (twi_write((addr<<1) | TWI_WRITE) == 0)
165 {
166 /* Send memory address */
167 if (twi_write(memaddr) == 0)
168 {
169 /* Send data byte */
170 if (twi_write(data) == 0)
171 {
172 twi_stop();
173 return 0; /* success (ACKs received) */
174 }
175 }
176 }
177 twi_stop();
178 return 1; /* failure (NACK or other error) */
179}
