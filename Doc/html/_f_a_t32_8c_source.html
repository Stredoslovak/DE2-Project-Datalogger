22//***************************************************************************
23//Function: to read data from boot sector of SD card, to determine important
24//parameters like bytesPerSector, sectorsPerCluster etc.
25//Arguments: none
26//return: none
27//***************************************************************************
28unsigned char getBootSectorData (void)
29{
30struct BS_Structure *bpb; //mapping the buffer onto the structure
31struct MBRinfo_Structure *mbr;
32struct partitionInfo_Structure *partition;
33unsigned long dataSectors;
34
35unusedSectors = 0;
36
37SD_readSingleBlock(0);
38bpb = (struct BS_Structure *)buffer;
39
40if(bpb->jumpBoot[0]!=0xE9 && bpb->jumpBoot[0]!=0xEB) //check if it is boot sector
41{
42 mbr = (struct MBRinfo_Structure *) buffer; //if it is not boot sector, it must be MBR
43
44 if(mbr->signature != 0xaa55) return 1; //if it is not even MBR then it's not FAT32
45
46 partition = (struct partitionInfo_Structure *)(mbr->partitionData);//first partition
47 unusedSectors = partition->firstSector; //the unused sectors, hidden to the FAT
48
49 SD_readSingleBlock(partition->firstSector);//read the bpb sector
50 bpb = (struct BS_Structure *)buffer;
51 if(bpb->jumpBoot[0]!=0xE9 && bpb->jumpBoot[0]!=0xEB) return 1;
52}
53
54bytesPerSector = bpb->bytesPerSector;
55//transmitHex(INT, bytesPerSector); transmitByte(' ');
56sectorPerCluster = bpb->sectorPerCluster;
57//transmitHex(INT, sectorPerCluster); transmitByte(' ');
58reservedSectorCount = bpb->reservedSectorCount;
59rootCluster = bpb->rootCluster;// + (sector / sectorPerCluster) +1;
60firstDataSector = bpb->hiddenSectors + reservedSectorCount + (bpb->numberofFATs * bpb->FATsize_F32);
61
62dataSectors = bpb->totalSectors_F32
63 - bpb->reservedSectorCount
64 - ( bpb->numberofFATs * bpb->FATsize_F32);
65totalClusters = dataSectors / sectorPerCluster;
66//transmitHex(LONG, totalClusters); transmitByte(' ');
67
68if((getSetFreeCluster (TOTAL_FREE, GET, 0)) > totalClusters) //check if FSinfo free clusters count is valid
69 freeClusterCountUpdated = 0;
70else
71 freeClusterCountUpdated = 1;
72return 0;
73}

85//***************************************************************************
86//Function: get cluster entry value from FAT to find out the next cluster in the chain
87//or set new cluster entry in FAT
88//Arguments: 1. current cluster number, 2. get_set (=GET, if next cluster is to be found or = SET,
89//if next cluster is to be set 3. next cluster number, if argument#2 = SET, else 0
90//return: next cluster number, if if argument#2 = GET, else 0
91//****************************************************************************
92unsigned long getSetNextCluster (unsigned long clusterNumber,
93 unsigned char get_set,
94 unsigned long clusterEntry)
95{
96unsigned int FATEntryOffset;
97unsigned long *FATEntryValue;
98unsigned long FATEntrySector;
99unsigned char retry = 0;
100
101//get sector number of the cluster entry in the FAT
102FATEntrySector = unusedSectors + reservedSectorCount + ((clusterNumber * 4) / bytesPerSector) ;
103
104//get the offset address in that sector number
105FATEntryOffset = (unsigned int) ((clusterNumber * 4) % bytesPerSector);
106
107//read the sector into a buffer
108while(retry <10)
109{ if(!SD_readSingleBlock(FATEntrySector)) break; retry++;}
110
111//get the cluster address from the buffer
112FATEntryValue = (unsigned long *) &buffer[FATEntryOffset];
113
114if(get_set == GET)
115 return ((*FATEntryValue) & 0x0fffffff);
116
117
118*FATEntryValue = clusterEntry; //for setting new value in cluster entry in FAT
119
120SD_writeSingleBlock(FATEntrySector);
121
122return (0);
123}

125//********************************************************************************************
126//Function: to get or set next free cluster or total free clusters in FSinfo sector of SD card
127//Arguments: 1.flag:TOTAL_FREE or NEXT_FREE,
128// 2.flag: GET or SET
129// 3.new FS entry, when argument2 is SET; or 0, when argument2 is GET
130//return: next free cluster, if arg1 is NEXT_FREE & arg2 is GET
131// total number of free clusters, if arg1 is TOTAL_FREE & arg2 is GET
132// 0xffffffff, if any error or if arg2 is SET
133//********************************************************************************************
134unsigned long getSetFreeCluster(unsigned char totOrNext, unsigned char get_set, unsigned long FSEntry)
135{
136struct FSInfo_Structure *FS = (struct FSInfo_Structure *) &buffer;
137unsigned char error;
138SD_readSingleBlock(unusedSectors + 1);
139
140if((FS->leadSignature != 0x41615252) || (FS->structureSignature != 0x61417272) || (FS->trailSignature !=0xaa550000))
141 return 0xffffffff;
142
143 if(get_set == GET)
144 {
145 if(totOrNext == TOTAL_FREE)
146 return(FS->freeClusterCount);
147 else // when totOrNext = NEXT_FREE
148 return(FS->nextFreeCluster);
149 }
150 else
151 {
152 if(totOrNext == TOTAL_FREE)
153 FS->freeClusterCount = FSEntry;
154 else // when totOrNext = NEXT_FREE
155 FS->nextFreeCluster = FSEntry;
156
157 error = SD_writeSingleBlock(unusedSectors + 1); //update FSinfo
158 }
159 return 0xffffffff;
160}

163//Function: to get DIR/FILE list or a single file address (cluster number) or to delete a specified file

169struct dir_Structure* findFiles (unsigned char flag, unsigned char *fileName)
170{
171// uart_puts_P("findFiles: Started\r\n");
172unsigned long cluster, sector, firstSector, firstCluster, nextCluster;
173struct dir_Structure *dir;
174unsigned int i;
175unsigned char j;

179uart_puts_P("findFiles: Root cluster = ");
180transmitHex(LONG, cluster);
181uart_puts_P("\r\n");
182
183while(1)
184{
185 loopCount++;
186 if(loopCount > 100) { // ← Pridajte timeout
187 uart_puts_P("findFiles: Loop timeout!\r\n");
188 return 0;
189 }
190
191 uart_puts_P("findFiles: Reading cluster ");
192 transmitHex(LONG, cluster);
193 uart_puts_P("\r\n");
194
195 firstSector = getFirstSector (cluster);
196
197 for(sector = 0; sector < sectorPerCluster; sector++)
198 {
199 uart_puts_P("findFiles: Reading sector\r\n");
200 SD_readSingleBlock (firstSector + sector);
201 uart_puts_P("findFiles: Sector read\r\n");
202
203 for(i=0; i<bytesPerSector; i+=32)
204 {
205 dir = (struct dir_Structure *) &buffer[i];
206
207 if(dir->name[0] == EMPTY) //indicates end of the file list

209 uart_puts_P("findFiles: Empty entry found\r\n");
210 if(flag == DELETE)
211 uart_puts_P("File does not exist!");
212 return 0;
213 }
214 if((dir->name[0] != DELETED) && (dir->attrib != ATTR_LONG_NAME))
215 {
216 if((flag == GET_FILE) || (flag == DELETE))
217 {
218 for(j=0; j<11; j++)
219 if(dir->name[j] != fileName[j]) break;
220 if(j == 11)
221 {
222 if(flag == GET_FILE)
223 {
224 appendFileSector = firstSector + sector;
225 appendFileLocation = i;
226 appendStartCluster = (((unsigned long) dir->firstClusterHI) << 16) | dir->firstClusterLO;
227 fileSize = dir->fileSize;
228 return (dir);
229 }
230 else //when flag = DELETE
231 {
232 TX_NEWLINE;
233 uart_puts_P(("Deleting.."));
234 TX_NEWLINE;
235 TX_NEWLINE;
236 firstCluster = (((unsigned long) dir->firstClusterHI) << 16) | dir->firstClusterLO;
237
238 //mark file as 'deleted' in FAT table

243
244 //update next free cluster entry in FSinfo sector

246 if(firstCluster < cluster)
247 getSetFreeCluster (NEXT_FREE, SET, firstCluster);
248
249 //mark all the clusters allocated to the file as 'free'
250 while(1)
251 {
252 nextCluster = getSetNextCluster (firstCluster, GET, 0);
253 getSetNextCluster (firstCluster, SET, 0);
254 if(nextCluster > 0x0ffffff6)
255 {uart_puts_P("File deleted!");return 0;}
256 firstCluster = nextCluster;
257 }
258 }
259 }
260 }
261 else //when flag = GET_LIST
262 {
263 // TX_NEWLINE;
264 // uart_puts_P("JUST a TEST in listing\r\n");
265 for(j=0; j<11; j++)
266 {
267 if(j == 8) transmitByte(' ');
268 // transmitByte (dir->name[j]);
269 }
270 uart_puts_P(" ");
271 if((dir->attrib != 0x10) && (dir->attrib != 0x08))
272 {
273 uart_puts_P("FILE" );
274 uart_puts_P(" ");
275 displayMemory(LOW, dir->fileSize);
276 }
277 else
278 {
279 if (dir->attrib == 0x10)
280 {
281 uart_puts(("DIR" ));
282 }
283 else
284 {
285 uart_puts(("ROOT" ));
286 }
287 }
288 }
289 }
290 }
291 }
292
293 // uart_puts_P("findFiles: Getting next cluster...\r\n");
294 cluster = (getSetNextCluster (cluster, GET, 0));
295 // uart_puts_P("findFiles: Next cluster = ");
296 // transmitHex(LONG, cluster);
297 // uart_puts_P("\r\n");
298
299 if(cluster > 0x0ffffff6) {
300 // uart_puts_P("findFiles: End of cluster chain\r\n");
301 return 0;
302 }
303 if(cluster == 0) {
304 // uart_puts_P("Error in getting cluster");
305 return 0;
306 }
307}
308return 0;

311//***************************************************************************
312//Function: if flag=READ then to read file from SD card and send contents to UART
313//if flag=VERIFY then functions will verify whether a specified file is already existing
314//Arguments: flag (READ or VERIFY) and pointer to the file name
315//return: 0, if normal operation or flag is READ
316// 1, if file is already existing and flag = VERIFY; or if flag=READ and file does not exist
317// 2, if file name is incompatible
318//***************************************************************************
319unsigned char readFile (unsigned char flag, unsigned char *fileName)
320{
321struct dir_Structure *dir;
322unsigned long cluster, byteCounter = 0, fileSize, firstSector;
323unsigned int k;
324unsigned char j, error;
325
326uart_puts_P("readFile: Converting filename...\r\n");
327error = convertFileName (fileName); //convert fileName into FAT format
328if(error) {
329 uart_puts_P("readFile: Invalid filename!\r\n");
330 return 2;
331}
332
333uart_puts_P("readFile: Filename converted: ");
334for(j=0; j<11; j++) transmitByte(fileName[j]);
335uart_puts_P("\r\n");
336
337uart_puts_P("readFile: Calling findFiles...\r\n");
338dir = findFiles (GET_FILE, fileName); //get the file location
339uart_puts_P("readFile: findFiles returned\r\n");
340
341if(dir == 0)
342{
343 uart_puts_P("readFile: File not found\r\n");
344 if(flag == READ) return (1);
345 else return (0);
346}
347
348uart_puts_P("readFile: File found!\r\n");
349if(flag == VERIFY) return (1); //specified file name is already existing
350
351cluster = (((unsigned long) dir->firstClusterHI) << 16) | dir->firstClusterLO;
352
353fileSize = dir->fileSize;
354
355TX_NEWLINE;
356TX_NEWLINE;
357
358while(1)
359{
360 firstSector = getFirstSector (cluster);
361
362 for(j=0; j<sectorPerCluster; j++)
363 {
364 SD_readSingleBlock(firstSector + j);
365
366 for(k=0; k<512; k++)
367 {
368 // transmitByte(buffer[k]);
369 if ((byteCounter++) >= fileSize ) return 0;
370 }
371 }
372 cluster = getSetNextCluster (cluster, GET, 0);
373 if(cluster == 0) {uart_puts_P("Error in getting cluster"); return 0;}
374}
375return 0;
376}

378//***************************************************************************
379//Function: to convert normal short file name into FAT format
380//Arguments: pointer to the file name
381//return: 0, if successful else 1.
382//***************************************************************************
383unsigned char convertFileName (unsigned char *fileName)
384{
385unsigned char fileNameFAT[11];
386unsigned char j, k;
387
388// uart_puts_P("cvt: start\r\n");
389
390for(j=0; j<12; j++)
391 if(fileName[j] == '.') break;
392
393// uart_puts_P("cvt: dot at ");
394// transmitByte(j + '0');
395// uart_puts_P("\r\n");
396
397if(j>8) {
398 // uart_puts_P("cvt: name too long\r\n");
399 return 1;
400}
401
402for(k=0; k<j; k++)
403 fileNameFAT[k] = fileName[k];
404
405for(k=j; k<=7; k++)
406 fileNameFAT[k] = ' ';
407
408uart_puts_P("cvt: name done\r\n");
409
410j++;
411for(k=8; k<11; k++)
412{
413 if(fileName[j] != 0)
414 fileNameFAT[k] = fileName[j++];
415 else
416 while(k<11)
417 fileNameFAT[k++] = ' ';
418}
419
420uart_puts_P("cvt: ext done\r\n");
421
422for(j=0; j<11; j++)
423 if((fileNameFAT[j] >= 0x61) && (fileNameFAT[j] <= 0x7a))
424 fileNameFAT[j] -= 0x20;
425
426uart_puts_P("cvt: caps done\r\n");
427
428for(j=0; j<11; j++)
429 fileName[j] = fileNameFAT[j];
430
431uart_puts_P("cvt: OK\r\n");
432
433return 0;
434}

436//************************************************************************************
437//Function: to create a file in FAT32 format in the root directory if given

443{
444uart_puts_P("writeFile: Started\r\n");
445unsigned char j,k, data, error, fileCreatedFlag = 0, start = 0, appendFile = 0, sector=0;
446unsigned int i, firstClusterHigh=0, firstClusterLow=0;

451j = readFile (VERIFY, fileName);

455
456if(j == 1)

474 start = 1;
475}

478 uart_puts_P("writeFile: Invalid filename\r\n");

480}

541 }

560 }

564}

619 dir->timeTenth = 0; //always set to 0

621 dir->createDate = dateFAT; //setting date of file creation, obtained from RTC

628

630 fileCreatedFlag = 1;
631
632 //TX_NEWLINE;
633 //TX_NEWLINE;
634 //uart_puts_P(PSTR(" File Created! "));
635
636 freeMemoryUpdate (REMOVE, fileSize); //updating free memory count in FSinfo sector
637
638 }
639 }
640 }
641
642 cluster = getSetNextCluster (prevCluster, GET, 0);
643
644 if(cluster > 0x0ffffff6)
645 {
646 if(cluster == EOF) //this situation will come when total files in root is multiple of (32*sectorPerCluster)
647 {
648 cluster = searchNextFreeCluster(prevCluster); //find next cluster for root directory entries
649 getSetNextCluster(prevCluster, SET, cluster); //link the new cluster of root to the previous cluster
650 getSetNextCluster(cluster, SET, EOF); //set the new cluster as end of the root directory
651 }
652
653 else
654 {
655 uart_puts_P("End of Cluster Chain");
656 return 1;
657 }
658 }
659 if(cluster == 0) {uart_puts_P("Error in getting cluster"); return 1;}
660
661 prevCluster = cluster;
662 }
663
664 return 0;
665}

668//***************************************************************************
669//Function: to search for the next free cluster in the root directory
670// starting from a specified cluster
671//Arguments: Starting cluster
672//return: the next free cluster
673//****************************************************************
674unsigned long searchNextFreeCluster (unsigned long startCluster)
675{
676 unsigned long cluster, *value, sector;
677 unsigned char i;
678
679 startCluster -= (startCluster % 128); //to start with the first file in a FAT sector
680 for(cluster =startCluster; cluster <totalClusters; cluster+=128)
681 {
682 sector = unusedSectors + reservedSectorCount + ((cluster * 4) / bytesPerSector);
683 SD_readSingleBlock(sector);
684 for(i=0; i<128; i++)
685 {
686 value = (unsigned long *) &buffer[i*4];
687 if(((*value) & 0x0fffffff) == 0)
688 return(cluster+i);
689 }
690 }
691
692 return 0;
693}
