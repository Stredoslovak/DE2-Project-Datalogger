33{

122//******************************************************************
123//Function : to send a command to SD card
124//Arguments : unsigned char (8-bit command value)
125// & unsigned long (32-bit command argument)
126//return : unsigned char; response byte
127//******************************************************************
128unsigned char SD_sendCommand(unsigned char cmd, unsigned long arg)
129{
130
131unsigned char response, retry=0, status;
132// uart_puts_P("In SD_sendCommand\r\n");
133//SD card accepts byte address while SDHC accepts block address in multiples of 512
134//so, if it's SD card we need to convert block address into corresponding byte address by
135//multipying it with 512. which is equivalent to shifting it left 9 times
136//following 'if' loop does that
137
138if(SDHC_flag == 0)
139{
140 uart_puts_P("SDHC_flag == 0\r\n");
141if(cmd == READ_SINGLE_BLOCK ||
142 cmd == READ_MULTIPLE_BLOCKS ||
143 cmd == WRITE_SINGLE_BLOCK ||
144 cmd == WRITE_MULTIPLE_BLOCKS ||
145 cmd == ERASE_BLOCK_START_ADDR||
146 cmd == ERASE_BLOCK_END_ADDR )
147 {
148 arg = arg << 9;
149 }
150}
151SD_CS_ASSERT;
152uart_puts_P("After SD_CS_ASSERT\r\n");
153_delay_ms(1); //small delay before sending command
154SPI_transmit(cmd | 0x40); //send command, first two bits always '01'
155SPI_transmit(arg>>24);
156SPI_transmit(arg>>16);
157SPI_transmit(arg>>8);
158SPI_transmit(arg);
159// uart_puts_P("CMDs\r\n");
160if(cmd == SEND_IF_COND) //it is compulsory to send correct CRC for CMD8 (CRC=0x87) & CMD0 (CRC=0x95)
161 SPI_transmit(0x87); //for remaining commands, CRC is ignored in SPI mode
162else
163 SPI_transmit(0x95);
164
165while((response = SPI_receive()) == 0xff) //wait response
166 if(retry++ > 0xfe) break; //time out error
167
168if(response == 0x00 && cmd == 58) //checking response of CMD58
169{
170 status = SPI_receive() & 0x40; //first byte of the OCR register (bit 31:24)
171 if(status == 0x40) SDHC_flag = 1; //we need it to verify SDHC card
172 else SDHC_flag = 0;
173
174 SPI_receive(); //remaining 3 bytes of the OCR register are ignored here
175 SPI_receive(); //one can use these bytes to check power supply limits of SD
176 SPI_receive();
177}
178
179SPI_receive(); //extra 8 CLK
180SD_CS_DEASSERT;
181
182return response; //return state
183}

185//*****************************************************************
186//Function : to erase specified no. of blocks of SD card
187//Arguments : none
188//return : unsigned char; will be 0 if no error,
189// otherwise the response byte will be sent
190//*****************************************************************
191unsigned char SD_erase (unsigned long startBlock, unsigned long totalBlocks)
192{
193unsigned char response;
194
195response = SD_sendCommand(ERASE_BLOCK_START_ADDR, startBlock); //send starting block address
196if(response != 0x00) //check for SD status: 0x00 - OK (No flags set)
197 return response;
198
199response = SD_sendCommand(ERASE_BLOCK_END_ADDR,(startBlock + totalBlocks - 1)); //send end block address
200if(response != 0x00)
201 return response;
202
203response = SD_sendCommand(ERASE_SELECTED_BLOCKS, 0); //erase all selected blocks
204if(response != 0x00)
205 return response;
206
207return 0; //normal return
208}

216unsigned char SD_readSingleBlock(unsigned long startBlock)

245//******************************************************************
246//Function : to write to a single block of SD card
247//Arguments : none
248//return : unsigned char; will be 0 if no error,
249// otherwise the response byte will be sent
250//******************************************************************
251unsigned char SD_writeSingleBlock(unsigned long startBlock)
252{
253unsigned char response;
254unsigned int i, retry=0;
255
256 response = SD_sendCommand(WRITE_SINGLE_BLOCK, startBlock); //write a Block command
257
258 if(response != 0x00) return response; //check for SD status: 0x00 - OK (No flags set)
259
260SD_CS_ASSERT;
261
262SPI_transmit(0xfe); //Send start block token 0xfe (0x11111110)
263
264for(i=0; i<512; i++) //send 512 bytes data
265 SPI_transmit(buffer[i]);
266
267SPI_transmit(0xff); //transmit dummy CRC (16-bit), CRC is ignored here
268SPI_transmit(0xff);
269
270response = SPI_receive();
271
272if( (response & 0x1f) != 0x05) //response= 0xXXX0AAA1 ; AAA='010' - data accepted
273{ //AAA='101'-data rejected due to CRC error
274 SD_CS_DEASSERT; //AAA='110'-data rejected due to write error
275 return response;
276}
277
278while(!SPI_receive()) //wait for SD card to complete writing and get idle
279if(retry++ > 0xfffe){SD_CS_DEASSERT; return 1;}
280
281SD_CS_DEASSERT;
282SPI_transmit(0xff); //just spend 8 clock cycle delay before reasserting the CS line
283SD_CS_ASSERT; //re-asserting the CS line to verify if card is still busy
284
285while(!SPI_receive()) //wait for SD card to complete writing and get idle
286 if(retry++ > 0xfffe){SD_CS_DEASSERT; return 1;}
287SD_CS_DEASSERT;
288
289return 0;
290}
